Java集合总结

由于Java集合中有众多接口和实现类，本文只选取常用的集合。

Collection

Collection接口在java.util中，是所有单列集合的父接口，比如List和Set

常用方法：

  1. public boolean add(E e); 添加给定对象
  2. public void clear(); 清空集合
  3. public boolean remove(E e); 删除给定对象
  4. public boolean contains(E e); 判断集合中是否包含指定对象
  5. public boolean isEmpty(); 判断当前集合是否为空
  6. public int size(); 返回集合中元素的个数
  7. public Object[] toArray(); 将集合元素存储到数组中Collection集合在java.util中，是所有单列集合的父接口，比如List和Set
public class CollectionDemo {
    public static void main(String[] args) {

        // 创建集合对象
        Collection<String> coll = new ArrayList<>();

        // 添加元素
        coll.add("古力娜扎");
        coll.add("迪丽热巴");
        coll.add("马儿扎哈");
        System.out.println(coll);  // [古力娜扎, 迪丽热巴, 马儿扎哈]

        // 判断是否包含指定对象
        boolean b = coll.contains("马儿扎哈");
        System.out.println(b);  // true

        // 删除指定对象
        coll.remove("迪丽热巴");
        System.out.println(coll);  // [古力娜扎, 马儿扎哈]

        // 判断集合是否为空
        boolean e = coll.isEmpty();
        System.out.println(e);  // false

        // 集合中的元素个数
        int size = coll.size();
        System.out.println(size);  // 2

        // 将集合元素存储到数组
        Object[] obj = coll.toArray();
        for (int i = 0; i < obj.length; i++) {
            System.out.println(obj[i]); // 依次输出古力娜扎 马儿扎哈
        }

        // 清空集合中的所有元素，但是不删除集合
        coll.clear();
        System.out.println(coll); // []
    }
}

List

List集合接口中有四个常用的方法 1. add() 2. remove() 3. get() 4. set()

public class ListDemo {
    public static void main(String[] args) {

        // 创建一个list集合
        List<String> list = new ArrayList<>();

        // 添加元素
        list.add("a");
        list.add("d");
        System.out.println(list);  // [a, d]

        // 在指定位置添加元素
        list.add(2, "m");
        System.out.println(list);  // [a, d, m]

        // 移除指定位置的元素
        list.remove(2);
        System.out.println(list);  // [a, d]

        // 用指定元素替换指定位置的元素
        list.set(1, "c");
        System.out.println(list);  // [a, c]

        // 获取指定位置的元素
        String s = list.get(1);
        System.out.println(s);  // c
    }
}

ArrayList

ArrayList是变长数组，类似于C++中的vector。注意<>中是泛型，只能是引用类型，而不能是基本类型

基本类型 包装类（引用类型） byte Byte short Short int Integer 需要记 long Long float Float double Double char Character 需要记 boolean Boolean

常用方法有：

  1. public boolean add(E e); 添加元素
  2. public E get(int index); 读取元素
  3. public E remove(int index); 删除元素
  4. public int size(); 获取集合中元素个数
public class ArrayListDemo1 {
    public static void main(String[] args) {

        // 创建ArrayList集合
        ArrayList<String> list = new ArrayList<>();
        ArrayList<Integer> list1 = new ArrayList<>();

        // 向集合中添加一些数据，采用add方法
        list.add("马儿扎哈");
        list.add("interesting");
        list.add("123456");
        list1.add(100);
        list1.add(200);
        System.out.println(list);  // [马儿扎哈, interesting, 123456]
        System.out.println(list1); // [100, 200]

        // 读取下标为1的元素
        String get_s = list.get(1);
        int get_i = list1.get(1);
        System.out.println(get_s);  // interesting
        System.out.println(get_i);  // 200

        // 从集合中移除下标为0的元素
        String remove_s = list.remove(0);
        System.out.println(remove_s);  // 马儿扎哈
        System.out.println(list);  // [interesting, 123456]

        // 获取集合中的元素个数
        int size = list.size();
        System.out.println(size);  // 2

        // 遍历集合,IDEA中用list.fori快速生成
        for(int i = 0; i < list.size(); i++){
            System.out.println(list.get(i));  // interesting 123456
        }
    }
}

LinkedList

LinkedList是List接口的另一个实现类

常用的方法： 一、与添加相关的方法

  1. public void addFirst(E e); 添加指定元素到链表的开头
  2. public void addLast(E e); 添加指定的元素到链表的结尾
  3. public void push(E e); 添加元素到此列表所表示的堆栈，就是添加元素到链表开头

二、与获取相关的方法

  1. public E getFirst(); 获取链表的第一个元素
  2. public E getLast(); 获取链表的最后一个元素

三、 删除相关的方法

  1. public E removeFirst(); 移除链表的第一个元素
  2. public E removeLast(); 移除链表的最后一个元素
  3. public E pop(); 列表所表示的堆栈中弹出一个元素， 就是弹出链表开头的元素
public class LinkedListDemo {
    public static void main(String[] args) {

        // 创建LinkedList集合对象
        LinkedList<String> linkedlist = new LinkedList<>();

        // 添加元素,可以使用其父类的方法添加
        linkedlist.add("love");
        linkedlist.add("you");
        System.out.println(linkedlist);  // [love, you]

        // 在链表开头添加元素
        linkedlist.addFirst("I");
        System.out.println(linkedlist);  // [I, love, you]

        // 在链表结尾添加元素, 等效于add方法
        linkedlist.addLast("!");
        System.out.println(linkedlist);  // [I, love, you, !]

        // 在链表所表示的堆栈上添加，等效于addFirst,在链表开头添加一个元素
        linkedlist.push("Hello");
        System.out.println(linkedlist);  // [Hello, I, love, you, !]

        // 获取并返回第一个元素
        String first = linkedlist.getFirst();
        System.out.println(first);  // Hello

        // 获取并返回最后一个元素
        String last = linkedlist.getLast();
        System.out.println(last);  // !

        // 移除第一个元素
        String s = linkedlist.removeFirst();
        System.out.println(s);  // Hello
        System.out.println(linkedlist);  // [I, love, you, !]

        // 移除最后一个元素
        String s1 = linkedlist.removeLast();
        System.out.println(s1);  // !
        System.out.println(linkedlist);  // [I, love, you]

        // 弹出一个元素，相当于removeFirst， 弹出第一个元素
        String s2 = linkedlist.pop();
        System.out.println(s2);  // I
        System.out.println(linkedlist);  // [love, you]

        // 判断链表是否为空
        boolean empty = linkedlist.isEmpty();
        System.out.println(empty);  // false
    }
}

HashSet

HashSet是Set接口的实现类

Person类会默认继承Object类的hashCode方法，该方法返回对象的hash值

public class HashSetDemo {
    public static void main(String[] args) {
        // 创建HashSet集合
        Set<Integer> set = new HashSet<>();

        // 添加元素
        set.add(1);
        set.add(2);
        set.add(4);
        set.add(3);
        set.add(3);

        // 遍历set集合
        Iterator<Integer> iterator = set.iterator();
        while(iterator.hasNext()){
            Integer n = iterator.next();
            System.out.println(n);  // 依次输出1， 2， 3， 4
        }
        //  增强for循环
        for (Integer integer : set) {
            System.out.println(integer );  // 依次输出1， 2， 3，4
        }

        // 返回对象的hash值
        Person p1 = new Person();
        int hash1 = p1.hashCode();
        System.out.println(hash1);  // 284720968
    }
}

LinkedHashSet

LinkedHashSet是HashSet类的子类

和HashSet不同的是LinkedHashSet是有顺序的,按照存储的顺序

public class LinkedHashSetDemo {
    public static void main(String[] args) {

        // 创建一个HashSet
        HashSet<String> set = new HashSet<>();
        set.add("www");
        set.add("abc");
        set.add("abc");
        set.add("name");
        System.out.println(set);  // [abc, www, name]

        // 创建一个LinkedHashSet
        HashSet<String> linked = new LinkedHashSet<>();
        linked.add("www");
        linked.add("abc");
        linked.add("abc");
        linked.add("name");
        System.out.println(linked);  // [www, abc, name]
    }
}

Map

Map<K,V>是与Collection完全不同的两个接口，key不允许重复，而value允许重复

常用方法：

  1. public V put(K key, V value); 把指定的键值对添加到集合中
  2. public V remove(Object key); 把指定的键对应的键值对从集合中删除，返回删除的元素的值
  3. public V get(Object key); 根据指定的键获取对应的值
  4. boolean containsKey(Object key); 判断集合中是否包含指定的键
  5. boolean Set<K> keySet(); 获取集合中的所有键，存储到Set集合中
  6. public Set<Map.Entry<K,V>> entrySet(); 获取map集合中的所有键值对，存储到Set集合中
public class MapDemo {
    public static void main(String[] args) {

        // 创建一个map集合
        Map<String, String> map = new HashMap<>();

        // 添加键值对
        map.put("李晨", "范冰冰");
        map.put("杨过", "小龙女");
        map.put("尹志平", "小龙女");
        System.out.println(map);  // {杨过=小龙女, 尹志平=小龙女, 李晨=范冰冰}

        // 根据指定的键删除对应的键值对
        String v1 = map.remove("杨过");
        System.out.println(v1);  // 小龙女
        System.out.println(map);  // {尹志平=小龙女, 李晨=范冰冰}

        // 根据键获取对应的值
        String v2 = map.get("李晨");
        System.out.println(v2);  // 范冰冰

        // 判断集合中是否包含指定的键
        boolean v3 = map.containsKey("尹志平");
        System.out.println(v3);  // true

        // 获取集合中所有的键存储到Set集合中
        Set<String> str = map.keySet();
        System.out.println(str);  // [尹志平, 李晨]

        // 遍历map
        for(String key : map.keySet()){
            System.out.println(map.get(key));  // 依次输出 小龙女 范冰冰
        }

        // 获取所有键值对，存储到集合中
        Set<Map.Entry<String, String>> ent = map.entrySet();
        System.out.println(ent);  // [尹志平=小龙女, 李晨=范冰冰]

        // 上面的entry对象存储在set中，有两个，entry对象可以分别调用getKey和getValue方法
        Iterator<Map.Entry<String, String>> it = ent.iterator();
        while(it.hasNext()){
            Map.Entry<String, String> next = it.next();
            System.out.println(next.getKey() + ":" + next.getValue());  
            // 依次输出 尹志平:小龙女 李晨:范冰冰
        }
    }
}

常用类：String

特点： 1. 字符串是常量，在创建后不能更改 2. 字符串在效果上面相当于char[]字符数组，但是底层原理是byte[]字节数组 创建字符串的3+1种方式(三种构造方法一种直接创建) 1. public String(); 创建一个空白字符串 2. public String(char[] array); 根据字符数组，创建对应的字符串 3. public String(byte[] array); 根据字节数组，创建对应的字符串 6. String str = "hello"; 直接创建

常用方法有：

一、比较相关方法

  1. public boolean equals(Object obj); 比较两个字符串内容
  2. public boolean equalsIgnoreCase(String str); 忽略大小写，进行内容比较

二、与获取相关的方法

  1. public int length(); 获取字符串中含有的字符个数
  2. public String concat(String str); 将当前字符串和参数字符串进行拼接成为新的字符串
  3. public char charAt(int index); 获取指定索引位置的单个字符
  4. int indexOf(String str); 查找参数字符串在本字符当中首次出现的索引位置，没有返回-1

三、截取相关的方法

  1. public String substring(int index); 从索引位置一直到字符串结尾，返回新字符串
  2. public String substring(int begin, int end); 截取索引范围内的子串，[begin,end)，包含左边，不包含右边

四、转换相关的方法

  1. public char[] toCharArray(); 将当前字符串拆分为字符数组作为返回值
  2. public byte[] getBytes(); 获得当前字符串底层的字节数组
  3. public String replace(CharSequence oldString, CharSequence newString); 将所有oldString替换为newString

五、分割字符串的方法

  1. public String split(String regex); 按照参数的规则也就是正则表达式规则，将字符串切分为若干部分
public class StringDemo1 {
    public static void main(String[] args) {

        // 使用空参构造方法
        String str1 = new String();
        System.out.println("--" + str1 + "--");  // ----

        // 根据字符数组，创建字符串
        char[] charArray = {'A', 'B', 'C'};
        String str2 = new String(charArray);
        System.out.println(str2);  // ABC

        // 根据字节数组，创建字符串
        byte[] byteArray = {97, 98, 99};
        String str3 = new String(byteArray);
        System.out.println(str3);  // abc

        // 直接创建
        String str4 = "Hello";
        System.out.println(str4);  // Hello

        /*
        * 程序当中直接写上双引号字符串，在字符串常量池中，new的不在池中
        * 对于基本类型来说，==是进行值的比较
        * 对于引用类型来说，==是进行地址的比较
        * equals是对内容进行比较，推荐"xyz".equals(str),不推荐str.equals("xyz")
        * */
        String str5 = "xyz";
        String str6 = "xyz";
        char[] chars = {'x','y','z'};
        String str7 = new String(chars);
        System.out.println(str5 == str6);  // true
        System.out.println(str5 == str7);  // false
        System.out.println(str6 == str7);  // false
        System.out.println(str5.equals(str6));  // true
        System.out.println(str5.equals(str7));  // true
        System.out.println(str6.equals(str7));  // true
        System.out.println(str7.equals("xyz"));  // true
        System.out.println("xyz".equals(str7));  // true
        System.out.println("XYZ".equals(str7));  // false
        System.out.println("XYZ".equalsIgnoreCase(str7));  // true

        // 获取字符串的长度
        int len = str7.length();
        System.out.println(len);  // 3

        // 拼接字符串
        String str8 = str4.concat(str7);
        System.out.println(str4);  // Hello
        System.out.println(str7);  // xyz
        System.out.println(str8);  // Helloxyz

        // 获取指定索引位置的字符
        char ch = str8.charAt(1);
        System.out.println(ch);  // e

        // 查找参数字符串在原来字符串中第一次出现的位置
        String str9 = "really";
        int pos1 = "i really really love you".indexOf(str9);
        int pos2 = "me".indexOf(str9);
        System.out.println(pos1);  // 2
        System.out.println(pos2);  // -1

        // 截取子串
        String str10 = str9.substring(2);
        System.out.println(str10);  // ally
        String str11 = str9.substring(2,4);
        System.out.println(str11);  // al

        // 将字符串拆分为字符数组
        char[] chararrs = str9.toCharArray();
        System.out.println(chararrs[0]);  // r
        System.out.println(chararrs.length);  // 6

        // 将字符串转换为字节数组
        byte[] bytes = str9.getBytes();
        for (int i = 0; i < bytes.length; i++) {
            System.out.print(bytes[i] + ",");  // 114,101,97,108,108,121,
        }

        // 用新字符子串替换指定的字符子串
        String str12 = str9.replace('l', 'o');
        System.out.println(str12);  // reaooy

        // 按照正则表达式规则切割字符串
        String str13 = "Hello,my,friend";
        String[] array = str13.split(",");
        for (int i = 0; i < array.length; i++) {
            System.out.println(array[i]);  // 依次返回 Hello my friend
        }
    }
}

工具类：Collections

Collections是集合工具类，用于操作集合

常用方法：

  1. public static <T> boolean addAll(Collection<T> c, T...elements) 往集合中添加一些元素
  2. public static void shuffle(List<?> list) 打乱顺序
  3. public static <T> void sort(List<T> list) 将集合中元素按照默认规则排序
  4. public static <T> void sort(List<T> list, Comparator<? super T>) 将集合中元素按照指定规则排序
public class CollectionsDemo {
    public static void main(String[] args) {
        ArrayList<String> list = new ArrayList<>();

        // 在集合中一次性添加多个元素
        Collections.addAll(list, "a", "b", "c", "d");
        System.out.println(list);  // [a, b, c, d]

        // 打乱集合中的元素顺序
        Collections.shuffle(list);
        System.out.println(list);  // [b, d, a, c] 每次输出都不一样

        // 将集合中的元素按照默认的规则排序
        Collections.sort(list);
        System.out.println(list);  // [a, b, c, d]

        // 无法对person类对象进行排序
        ArrayList<Person> list2 = new ArrayList<>();
        list2.add(new Person("张三", 18));
        list2.add(new Person("李四", 20));
        list2.add(new Person("王五", 19));
        System.out.println(list2);  
        // [Person{name='张三', age=18}, Person{name='李四',                                    //  age=20}, Person{name='王五', age=19}]
        
        // 此时使用Collections.sort(list2);报错

        // 在Person类中重写compareTo方法后可以按照自己定义的规则排序， 即按照年龄升序排序
        // 重写排序规则，按照年龄升序排序
        //  @Override
        //  public int compareTo(Person o) {
        //      return this.getAge() - o.getAge();
        //  }
        Collections.sort(list2);
        System.out.println(list2);  
        // [Person{name='张三', age=18}, Person{name='王五',
        // age=19}, Person{name='李四', age=20}]

        // 利用接口comparator重写规则,匿名内部类的方式
        Collections.sort(list, new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                return o2.charAt(0) - o1.charAt(0);
            }
        });
        System.out.println(list);  // [d, c, b, a]
    }
}

public class MathDemo {
    public static void main(String[] args) {

        // 获取绝对值
        System.out.println(Math.abs(3.14));  // 3.14
        System.out.println(Math.abs(0));  // 0
        System.out.println(Math.abs(-2.5));  // 2.5

        // 向上取整
        System.out.println(Math.ceil(3.9));  // 4.0
        System.out.println(Math.ceil(3.1));  // 4.0
        System.out.println(Math.ceil(3.0));  // 3.0

        // 向下取整
        System.out.println(Math.floor(3.9));  // 3.0
        System.out.println(Math.floor(3.1));  // 3.0
        System.out.println(Math.floor(3.0));  // 3.0

        // 四舍五入
        System.out.println(Math.round(3.9));  // 4.0
        System.out.println(Math.round(3.1));  // 3.0
        System.out.println(Math.round(3.0));  // 3.0
    }
}
